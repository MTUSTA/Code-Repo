print("Quiz: Average of positive numbers ")

#Open a new file for input
file_input = open("input.txt", "r")
#Open a new file for output
file_output = open("output.txt", "w")
sum=0
count=0
for aline in file_input.readlines():
	list = aline.split('=')
	for i in list:
		i=int(i)
		if i>0:
			sum=sum+i
			count=count+1
	if count>0:
		print("Average:",sum/count)
		file_output.write("Average: "+str(sum/count)+"\n")
		sum=0
		count=0
	else:
		print("There is no positive number")
		#print("There is no positive number",file=file_output)
		file_output.write("There is no positive number"+"\n")

file_input.close()
file_output.close()
