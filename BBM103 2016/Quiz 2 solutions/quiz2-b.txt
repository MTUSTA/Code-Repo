print("Quiz: Average of numbers divisible by 3 ")

#Open a new file for input
file_input = open("input.txt", "r")
#Open a new file for output
file_output = open("output.txt", "w")
sum=0
count=0
for aline in file_input.readlines():
	list = aline.split(',')
	for i in list:
		i=int(i)
		if i%3==0:
			sum=sum+i
			count=count+1
	if count>0:
		print("Average: ",sum/count)
		#print("Average: ",sum/count,file=file_output)
		file_output.write("Average: "+str(sum/count)+"\n")
		sum=0
		count=0
	else:
		print("There is no number divisible by 3")
		file_output.write("There is no number divisible by 3"+"\n")
		#print("There is no number which can be divided to 3",file=file_output)

file_input.close()
file_output.close()
